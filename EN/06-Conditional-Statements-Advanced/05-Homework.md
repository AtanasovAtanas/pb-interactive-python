[code-task title="Product of 3 Numbers" executionStrategy="python-code" requiresInput]
[code-editor language=python]
```
n1 = int(input())
n2 = int(input())
n3 = int(input())
if n1 == 0 or n2 == 0 or n3 == 0:
  print("zero")
else:
  negative_numbers = 0
  if n1 < 0: negative_numbers += 1
  if n2 < 0: negative_numbers += 1
  if n3 < 0: negative_numbers += 1
  if negative_numbers % 2 == 0:
    print("positive")
  else:
    print("negative")
```

```
    totalPrice = days * 10 * 0.80

    totalPrice = days * 50
```

```
num1 = int(input())

operation = input()
result = 0
if operation == "+":
  result = num1 + num2
elif operation == "-":
  result = num1 - num2
elif operation == "/":
  result = num1 / num2
elif operation == "*":
  result = num1 * num2
```
